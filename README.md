OPÇÃO COM FAÇA ... ENQUANTO

programa

{

   funcao inicio(){

      inteiro valor

      valor = 1

      faca {

         escreva (valor, "\n")

         valor = valor + 1

      }enquanto(valor <= 10)

   }

}

OPÇÃO COM PARA

programa

{

   funcao inicio(){

      para (inteiro valor = 1; valor <= 10; valor++){

         escreva (valor, "\n")

      }

   }

}
